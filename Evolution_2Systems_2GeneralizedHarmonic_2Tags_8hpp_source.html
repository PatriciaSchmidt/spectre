19 /*!
20  * \brief Conjugate momentum to the spacetime metric.
21  *
22  * \details If \f$ \psi_{ab} \f$ is the spacetime metric, and \f$ N \f$ and
23  * \f$ N^i \f$ are the lapse and shift respectively, then we define
24  * \f$ \Pi_{ab} = -\frac{1}{N} ( \partial_t \psi_{ab} + N^{i} \phi_{iab} ) \f$
25  * where \f$\phi_{iab}\f$ is the variable defined by the tag Phi.
26  */

28 struct Pi : db::SimpleTag {
29  using type = tnsr::aa<DataVector, Dim, Frame>;
30  static std::string name() noexcept { return "Pi"; }
31 };

204 /*!
205  * \ingroup OptionTagsGroup
206  * \brief Gauge control parameter to specify the spatial weighting function
207  * that enters damped harmonic gauge source function.
208  *
209  * \details The evolution gauge source function is multiplied by a spatial
210  * weight function which controls where and how much it sources the damped
211  * harmonic gauge equation. The weight function is:
212  * \f$ W(x^i) = \exp[-(r/\sigma_r)^2] \f$.
213  * This weight function can be written with an extra factor inside the exponent
214  * in literature, e.g. \cite Deppe2018uye. We will absorb it here in
215  * \f$\sigma_r\f$. The parameter this tag tags is \f$ \sigma_r \f$.
216  */
217 template <typename Frame>
218 struct GaugeHSpatialWeightDecayWidth : db::SimpleTag {
219  using type = double;
220  static std::string name() noexcept { return "GaugeHDecayWidth"; }
221  static constexpr OptionString help{
222  "Spatial width of weighting factor in evolution gauge"};
223 };
